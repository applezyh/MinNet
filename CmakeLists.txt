cmake_minimum_required(VERSION 3.22)

project(minnet_demo)

set(CXX_STANDARD 2A)
set(CXX_STANDARD_REQUIRE TRUE)

# find opencv
set (OpenCV_DIR D:\\opencv\\build)
find_package(OpenCV REQUIRED)
message(STATUS "OpenCV library status:")
message(STATUS "	config: ${OpenCV_DIR}")
message(STATUS "	version: ${OpenCV_VERSION}")
message(STATUS "	librarier: ${OpenCV_LIBS}")
message(STATUS "	include path: ${OpenCV_INCLUDE_DIRS}")

# add opencv include path
include_directories(${OpenCV_INCLUDE_DIRS})

add_subdirectory(src)
include_directories(./include)

add_executable( minnet_demo MinNet.cpp )
target_link_libraries( minnet_demo MINNET )
target_link_libraries( minnet_demo ${OpenCV_LIBS} )
